# construye nueva imagen con tag ipfs-api ( en lugar de latest)
docker build -t ipfs-api .
# listado de imagenes
docker images
# borrar una imagen
docker rmi id_image
# Ejecuta conetnedor a partir de la imagen pasada como parametro y con el nombre name 
docker stop ipfs-api
docker remove ipfs-api
docker run -d --name ipfs-api -p 8060:8060 ipfs-api
# Listado de contenedores en ejecución
docker ps

# Listado de contenedores activos
docker ps -a 
# borrar contenedor 
docker remove nombre_contenedor || container_id


# ver logs
docker logs -f <container_name_or_id>

# ver ip de interfaz docker
ip addr show docker0 | grep "inet" | awk '{print $2}' | cut -d'/' -f1

#open term
docker exec -it <container_id> /bin/bash

#instalar telnet
apt-get update && apt-get install -y inetutils-telnet
